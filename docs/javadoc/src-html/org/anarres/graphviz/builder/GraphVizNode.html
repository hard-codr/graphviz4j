005package org.anarres.graphviz.builder;
006
007import java.util.LinkedHashMap;
008import java.util.Map;
009import javax.annotation.CheckForNull;
010import javax.annotation.Nonnull;
011
012/**
013 *
014 * @author shevek
015 */
016public class GraphVizNode extends GraphVizObject<GraphVizNode> {
017
018    /** Use GraphVizAttribute. */
019    @Deprecated
020    public static final String ATTR_SHAPE = "shape";
021    /** Use GraphVizAttribute. */
022    @Deprecated
023    public static final String ATTR_HREF = "href";
024
025    private Map<GraphVizPort.Key, GraphVizPort> ports;
026    private int counter = 0;
027
028    /* pp */ GraphVizNode(@Nonnull GraphVizGraph graph, @Nonnull Key key, int id) {
029        super(graph, key, "n" + id);
030    }
031
032    @CheckForNull
033    public String getShape() {
034        return getAttribute(GraphVizAttribute.shape);
035    }
036
037    @Nonnull
038    public GraphVizNode shape(@CheckForNull String shape) {
039        return attr(GraphVizAttribute.shape, shape);
040    }
041
042    @Nonnull
043    public GraphVizPort port(@Nonnull Object object) {
044        GraphVizPort.Key key = new GraphVizPort.Key(getScope(), object);
045        if (!getGraph().isScopeVisible(key.getScope()))
046            return new GraphVizPort(this, key, -1);
047        if (ports == null)
048            ports = new LinkedHashMap<GraphVizPort.Key, GraphVizPort>();
049        GraphVizPort port = ports.get(key);
050        if (port == null) {
051            int id = counter++;
052            port = new GraphVizPort(this, key, id);
053            ports.put(key, port);
054        }
055        return port;
056    }
057
058    @CheckForNull
059    public String toLabelString() {
060        GraphVizLabel label = getLabel();
061        if (ports == null || ports.isEmpty()) {
062            if (label == null)
063                return null;
064            return label.toString();
065        }
066        GraphVizRecordLabel recordLabel = new GraphVizRecordLabel();
067        if (label != null)
068            recordLabel.title(label.getBuffer().toString());
069        for (Map.Entry<Key, GraphVizPort> e : ports.entrySet())
070            recordLabel.field(e.getValue());
071        return recordLabel.toString();
072    }
073}
