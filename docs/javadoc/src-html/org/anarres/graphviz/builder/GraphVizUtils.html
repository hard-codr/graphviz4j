006package org.anarres.graphviz.builder;
007
008import com.google.common.base.Preconditions;
009import com.google.common.escape.Escaper;
010import com.google.common.escape.Escapers;
011import java.io.File;
012import java.io.IOException;
013import javax.annotation.CheckForNull;
014import javax.annotation.Nonnull;
015
016/**
017 * Easy utilities for emitting GraphViz graphs.
018 *
019 * @author shevek
020 */
021public class GraphVizUtils {
022
023    public static final Escaper ESCAPE_HTML = Escapers.builder()
024            .addEscape('\\', "\\\\")
025            .addEscape('\"', "\\\"")
026            .addEscape('{', "\\{")
027            .addEscape('}', "\\}")
028            .addEscape('<', "&lt;")
029            .addEscape('>', "&gt;")
030            .addEscape('&', "&amp;")
031            .addEscape('\n', "<BR/>")
032            .addEscape('\r', "")
033            .build();
034
035    /* pp */ static int hashCode(@CheckForNull Object o) {
036        return o == null ? 0 : o.hashCode();
037    }
038
039    /* pp */ static boolean equals(@CheckForNull Object a, @CheckForNull Object b) {
040        if (a == b)
041            return true;
042        if (a == null)
043            return false;
044        return a.equals(b);
045    }
046
047    /**
048     * This supports a one-liner: GraphVizUtils.toGraphVizGraph(object).writeTo(new File(...)).
049     *
050     * @param object The object to graph.
051     * @return A newly constructed GraphVizGraph.
052     */
053    @Nonnull
054    public static GraphVizGraph toGraphVizGraph(@Nonnull GraphVizable object) {
055        Preconditions.checkNotNull(object, "GraphVizable object was null.");
056        GraphVizGraph graph = new GraphVizGraph();
057        object.toGraphViz(graph);
058        return graph;
059    }
060
061    /**
062     * Equivalent to {@link #toGraphVizGraph(GraphVizable) toGraphVizGraph}(object).{@link GraphVizGraph#writeTo(java.io.File) writeTo}(file).
063     *
064     * @param file The file to which to write the graph.
065     * @param object The object to convert.
066     * @throws IOException If an I/O error occurs.
067     */
068    public static void toGraphVizFile(@Nonnull File file, @Nonnull GraphVizable object) throws IOException {
069        toGraphVizGraph(object).writeTo(file);
070    }
071}
